Handler factory: export function createUrunYoneticisiHandlers(supabase: SupabaseClient<Database>): Map<string, HandlerFn>

Handlers:

'analyze_catalog':
  limit = typeof input.limit === 'number' ? Math.min(input.limit, 20) : 10
  period_days = typeof input.period_days === 'number' ? input.period_days : 30
  since_date = new Date(Date.now() - period_days * 86400000).toISOString()
  Query order_items with JOIN: supabase.from('order_items').select('product_id, quantity, unit_price, order:orders!inner(company_id, created_at)').eq('order.company_id', context.companyId).gte('order.created_at', since_date)
  If join is complex, fallback: query orders WHERE company_id = context.companyId AND created_at >= since_date, get order IDs, then query order_items WHERE order_id IN (orderIds)
  Aggregate by product_id in JavaScript: sum quantity (total_sold) and sum (quantity * unit_price) (total_revenue)
  Sort by total_revenue DESC, take top {limit}
  Return JSON.stringify with product_id, total_sold, total_revenue per top product

'suggest_pricing':
  Advisory-only — NO DB writes
  If product_id provided: query products WHERE id = product_id AND company_id = context.companyId for current price
  Also query dealer_prices WHERE product_id = product_id (if exists) for group overrides
  strategy = typeof input.strategy === 'string' ? input.strategy : 'competitive'
  Return formatted Turkish advisory string: 'Fiyat Stratejisi Onerisi ({strategy}): [analysis]. Tavsiye edilen fiyat araligi: mevcut fiyata gore +/- %10-20.'

'analyze_requests':
  limit = typeof input.limit === 'number' ? Math.min(input.limit, 20) : 10
  Query product_requests WHERE company_id = context.companyId (product_requests table from Phase 7)
  SELECT product_name, category, count(*) — if the table has these fields, otherwise select all and aggregate in JS
  Order by created_at DESC, limit to limit
  Return JSON.stringify of results

Handler factory: export function createSatinAlmaHandlers(supabase: SupabaseClient<Database>): Map<string, HandlerFn>

Handlers:

'create_purchase_order':
  items = Array.isArray(input.items) ? input.items : []
  confirmed = input.confirmed === true
  if (!confirmed) {
    return 'Siparis ozeti:\nKalemler: ' + JSON.stringify(items) + '\nOnaylamak icin: confirmed=true ile tekrar cagirin.'
  }
  INSERT into purchase_orders using (supabase as any):
  { company_id: context.companyId, supplier_id: typeof input.supplier_id === 'string' ? input.supplier_id : null, status: 'draft', items: items, notes: typeof input.notes === 'string' ? input.notes : null }
  On error: return '[Hata: ' + error.message + ']'
  Return 'Satin alma siparisi olusturuldu. ID: ' + data.id

'suggest_restock':
  limit = typeof input.limit === 'number' ? Math.min(input.limit, 20) : 10
  Query products WHERE company_id = context.companyId AND is_active = true AND stock_quantity <= low_stock_threshold
  (If low_stock_threshold doesn't exist as column, use: stock_quantity < 10 as fallback)
  Select: id, name, stock_quantity, sku
  Order by stock_quantity ASC, limit to limit
  Return JSON.stringify with restock suggestions including: product name, current stock, suggested_restock_quantity (= 50 - stock_quantity or similar simple formula)

Export: export const iadeKaliteTools: Tool[] = [manageReturnTool, trackComplaintTool]

Handler factory: export function createIadeKaliteHandlers(supabase: SupabaseClient<Database>): Map<string, HandlerFn>

Handlers:

'manage_return':
  confirmed = input.confirmed === true
  if (!confirmed) {
    return 'Iade ozeti:\nNeden: ' + String(input.reason ?? '') + '\nKalemler: ' + JSON.stringify(input.items ?? []) + '\nOnaylamak icin: confirmed=true ile tekrar cagirin.'
  }
  INSERT into return_requests using (supabase as any):
  { company_id: context.companyId, dealer_id: String(input.dealer_id), order_id: typeof input.order_id === 'string' ? input.order_id : null, reason: String(input.reason), status: 'pending', items: Array.isArray(input.items) ? input.items : [] }
  On error: return '[Hata: ' + error.message + ']'
  Return 'Iade talebi olusturuldu. ID: ' + data.id

'track_complaint':
  dealer_id = String(input.dealer_id ?? '')
  if (typeof input.description === 'string' && input.description.length > 0) {
    // Create mode
    complaint_type = typeof input.complaint_type === 'string' ? input.complaint_type : 'other'
    INSERT into quality_complaints using (supabase as any):
    { company_id: context.companyId, dealer_id, complaint_type, description: input.description, status: 'open' }
    On error: return '[Hata: ' + error.message + ']'
    Return 'Sikayet kaydedildi.'
  } else {
    // List mode
    Query (supabase as any).from('quality_complaints').select('*').eq('company_id', context.companyId).eq('dealer_id', dealer_id).order('created_at', { ascending: false }).limit(10)
    Return JSON.stringify(data ?? [])
  }
  </action>
