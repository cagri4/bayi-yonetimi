'log_visit':
  1. Validate dealer_id and actual_date
  2. INSERT into dealer_visits using (supabase as any):
     { company_id: context.companyId, dealer_id: String(input.dealer_id), planned_date: String(input.actual_date), actual_date: String(input.actual_date), visit_type: typeof input.visit_type === 'string' ? input.visit_type : 'routine', outcome: typeof input.outcome === 'string' ? input.outcome : null, notes: typeof input.notes === 'string' ? input.notes : null }
  3. On error return '[Hata: ' + error.message + ']'
  4. Return 'Ziyaret kaydedildi: {dealer_id} - {actual_date}' + (outcome ? ' - ' + outcome : '')

   description: 'Mevcut kampanya ve segment verilerine dayanarak yeni kampanya onerisi sunar. Sadece danismanlik — veritabanina yazmaz.'
   input_schema: { type: 'object', properties: { target_segment: { type: 'string', description: 'Hedef segment (yuksek | orta | dusuk | hepsi)' }, campaign_type: { type: 'string', description: 'Kampanya turu onerisi (indirim | sadakat | yeni_urun)' } }, required: [] }

Handler factory: export function createPazarlamaciHandlers(supabase: SupabaseClient<Database>): Map<string, HandlerFn>

Handlers:

'analyze_campaigns':
  Query campaigns WHERE company_id = context.companyId, ORDER by created_at DESC, limit to input.limit (default 5)
  Select: id, name, description, start_date, end_date, is_active
  Also query orders WHERE company_id = context.companyId to get order counts (simple count)
  Return JSON.stringify({ campaigns: campaignData, total_orders_in_period: orderCount })

'segment_dealers':
  period_days = typeof input.period_days === 'number' ? input.period_days : 90
  since_date = new Date(Date.now() - period_days * 86400000).toISOString()
  Query orders WHERE company_id = context.companyId AND created_at >= since_date
  Group by dealer_id in JavaScript, calculate total_amount per dealer
  Segment: high (total > 10000), medium (total 1000-10000), low (total < 1000)
  Also query dealers WHERE company_id = context.companyId to get dealers with 0 orders (inactive)
  Return JSON.stringify with segment counts and sample dealer IDs per segment

'suggest_campaign':
  Advisory-only tool — NO DB writes
  target_segment = typeof input.target_segment === 'string' ? input.target_segment : 'hepsi'
  campaign_type = typeof input.campaign_type === 'string' ? input.campaign_type : 'indirim'
  Return a formatted Turkish string with campaign suggestion based on inputs:
  Example: 'Kampanya Onerisi: [target_segment] segmentine yonelik [campaign_type] kampanyasi. Onerilen sure: 2 hafta. Beklenen etki: Siparis sayisinda %10-20 artis.'
